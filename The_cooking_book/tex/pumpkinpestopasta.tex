% Complete recipe example
\begin{recipe}
[% 
    preparationtime = {\unit[40]{min}},
    portion = {\portion{2}}
]
{Pumpkin Pesto Pasta}
    
    \graph
    {% pictures
        small=pic/pumpkinpestopasta2,     % small picture 890,950
        big=pic/pumpkinpestopasta % big picture 1263, 780
    }
    
    \introduction{%
        Pumpkins are great.
    }
    
    \ingredients{%
        2 carrots  \\
        a piece of hokkaido pumpkin\\        
        1 onion \\
        200g pasta per person \\  
        some frying oil\\ 
        green pesto \\    
        seasoning
    }
    
    \preparation{%
        \step Set water to boil and boil the pasta
        \step Meanwhile cut the onion into pieces, slice the carrots, and cut the pumpkin into pieces.
        \step Fry first the carrots, then add the pumpkin and onion and season.
        \step Mix the done pasta with pesto and add the vegetable on top.
    }
    
    \suggestion[]
    {%
        You can eat the peel of the hokkaido, that saves a lot of time.You can also just cook half the portion if you're alone. But better cook more and finish that pesto somehow! This is vegan, if you get a cheaper (vegan) pesto
    }
    
    
    \hint{ \#for1 \#fancy \#vegan   
    }
    
\end{recipe}
