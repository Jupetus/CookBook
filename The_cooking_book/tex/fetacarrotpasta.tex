% Complete recipe example
\begin{recipe}
[% 
    preparationtime = {\unit[20]{min}},
    portion = {\portion{1-2}}
]
{Feta Carrot Pasta}
    
    \graph
    {% pictures
        small=pic/fetacarrotpasta,     % small picture
        big=pic/fetacarrotpasta2 % big picture
    }
    
    \introduction{%
        A grateful dish if you have a grating device. If you have an automated grater, then it's even good for more people
    }
    
    \ingredients{%
        1-2 carrots per person \\
        50g feta cheese per person\\        
        1 onion \\
        100g pasta per person \\  
        some frying oil\\     
        pepper
    }
    
    \preparation{%
        \step Set water to boil and boil the pasta
        \step Meanwhile cut the onion into stripes and 
        grate the carrots 
        \step Fry first the onions, then add the carrots, add some pepper
        \step When they are done to your liking, add the pasta
        \step crumble the cheese and add it
    }
    
    \suggestion[]
    {%
        You can also add green pesto, champignons, ...
        I've tried roast onions, too. It adds a bit of crunch.
    }
    
    
    \hint{ If it is cold, it might pass as a salad. \#easy \#for1 
       
    }
    
\end{recipe}
